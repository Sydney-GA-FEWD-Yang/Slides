Variables

            var mark = $('#mark').val();

Now the conditionals


var result; //Unassigned at this point

if (mark < 50) {
    result = 'FL';
} else if (mark < 65) {
    result = 'PS';
} else if (mark < 75) {
    result = 'CR';
} else if (mark < 85) {
    result = 'DN';
} else {
    result = 'HD';
}

$('#result') = result; //Print result to screen

Reason 2: Grouping code/Creating formula

Calculate BMI = height/(weight^2)


            //BMI Index
function calculateBMI(height, weight){
    var heightToMeters = height / 100;
    var heightSquared = height * height;
    var BMI = weight / heightSquared;
    return BMI;
}

Using it as black box


var myHeightCM = 175;
var myWeightKG = 70;
var BMI = calculateBMI(myHeightCM, myWeightKG);

In this case, we don't care how BMI is calculated, we just trust the calculateBMI function.
